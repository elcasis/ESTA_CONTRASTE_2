    function setActiveExtSlot(slotId) {
      activeExtSlot = slotId;
      document.querySelectorAll('.extraction-slot').forEach(s => s.classList.remove('active-slot'));
      document.getElementById(`slot-ext-${slotId}`).classList.add('active-slot');

      // Generar el pool de opciones para rellenar
      const poolContainer = document.getElementById('pool-items-ext-container');
      poolContainer.innerHTML = '';

      // Valores correctos para la extracción
      let pool = [activeExercise.mu0, activeExercise.sigma, activeExercise.n, activeExercise.xbar];
      
      // Añadir la varianza (sigma2 o S2) como distractor si es diferente
      if (activeExercise.sigma2 && activeExercise.sigma2 !== activeExercise.sigma) {
        pool.push(activeExercise.sigma2);
      }

      // Añadir distractores aleatorios lógicos
      const mu0Num = parseFloat(activeExercise.mu0);
      const sigmaNum = parseFloat(activeExercise.sigma);
      const nNum = parseInt(activeExercise.n);
      const xbarNum = parseFloat(activeExercise.xbar);

      const d1 = Math.max(1, Math.round(mu0Num + (Math.random() > 0.5 ? 1 : -1) * (Math.floor(Math.random() * 5) + 3))).toString();
      const d2 = (sigmaNum + (Math.random() > 0.5 ? 1 : -1) * (Math.random() * 2 + 1.2)).toFixed(2).replace(/\.00$/, '');
      const d3 = Math.max(5, Math.round(nNum + (Math.random() > 0.5 ? 1 : -1) * (Math.floor(Math.random() * 15) + 8))).toString();
      const d4 = (xbarNum + (Math.random() > 0.5 ? 1 : -1) * (Math.random() * 2.5 + 1)).toFixed(2).replace(/\.00$/, '');

      [d1, d2, d3, d4].forEach(item => {
        if (!pool.includes(item)) pool.push(item);
      });

      // Mezclar aleatoriamente
      pool.sort(() => Math.random() - 0.5);

      pool.forEach(val => {
        const btn = document.createElement('button');
        btn.className = 'pool-item';
        btn.innerText = val;
        btn.onclick = () => selectExtValue(val);
        poolContainer.appendChild(btn);
      });
    }

    function selectExtValue(value) {

                        : activeExtSlot === 'sigma' ? activeExercise.sigma 
                        : activeExtSlot === 'n' ? activeExercise.n 
                        : activeExercise.xbar;

      const slotElement = document.getElementById(`slot-ext-${activeExtSlot}`);
      slotElement.className = 'extraction-slot';

      if (value === expectedVal) {
        slotElement.innerText = value;
        slotElement.classList.add('correct-slot', 'filled');
        extSelections[activeExtSlot] = value;

        let allFilled = Object.values(extSelections).every(v => v !== null);
        if (allFilled) {
          const feedback = document.getElementById('ext-feedback');
          feedback.style.display = 'block';
          feedback.className = "info-card green";
          document.getElementById('ext-feedback-text').innerText = `¡Excelente! Has extraído todos los datos necesarios: media histórica (μ₀ = ${activeExercise.mu0}), desviación estándar (${activeExercise.dist === 'Z' ? 'σ' : 'S'} = ${activeExercise.sigma}), tamaño muestral (n = ${activeExercise.n}) y media observada (xbar = ${activeExercise.xbar}).`;
          document.getElementById('btn-next-1').classList.remove('disabled');
          activeExtSlot = null;
          document.querySelectorAll('.extraction-slot').forEach(s => s.classList.remove('active-slot'));
          // Limpiar pool
          document.getElementById('pool-items-ext-container').innerHTML = '';
        } else {
          const order = ['mu0', 'sigma', 'n', 'xbar'];
          const nextSlot = order.find(k => extSelections[k] === null);

        }
      } else {
        slotElement.classList.add('wrong-slot');
        setTimeout(() => {
          slotElement.classList.remove('wrong-slot');
        }, 500);
      }
    }

    function checkHypothesisCorrectness() {
      let h0OpExpected = '≥';
      let h1OpExpected = '<';

      if (activeExercise.direction === 'right') {

    function setActiveMathSlot(slotId) {
      if (slotId === 'sigma') return; // Pre-rellenado
      activeMathSlot = slotId;
      document.querySelectorAll('.math-slot').forEach(s => s.classList.remove('active-slot'));
      document.getElementById(`slot-f-${slotId}`).classList.add('active-slot');

      // Generar el pool de opciones para rellenar
      const poolContainer = document.getElementById('pool-items-formula-container');
      poolContainer.innerHTML = '';

      // Crear pool con los valores correctos de xbar, mu0 y n
      let pool = [activeExercise.xbar, activeExercise.mu0, activeExercise.n];
      
      // Añadir distractores verdaderamente aleatorios
      const xbarNum = parseFloat(activeExercise.xbar);
      const mu0Num = parseFloat(activeExercise.mu0);
      const nNum = parseInt(activeExercise.n);

      // Distractor 1: Cerca de xbar (sumar o restar entre 1.5 y 4.5)
      const d1 = (xbarNum + (Math.random() > 0.5 ? 1 : -1) * (Math.random() * 3 + 1.5)).toFixed(2).replace(/\.00$/, '');
      
      // Distractor 2: Cerca de n (sumar o restar entre 10 y 30, asegurando que sea entero positivo)
      const d2 = Math.max(10, Math.round(nNum + (Math.random() > 0.5 ? 1 : -1) * (Math.floor(Math.random() * 20) + 10))).toString();
      
      // Distractor 3: Cerca de mu0 (sumar o restar entre 5 y 15)
      const d3 = Math.max(1, Math.round(mu0Num + (Math.random() > 0.5 ? 1 : -1) * (Math.floor(Math.random() * 10) + 5))).toString();
      
      // Distractor 4: Un número decimal aleatorio tipo Z crítico o p-valor común
      const commonStats = ["1.96", "1.645", "2.58", "0.05", "0.01", "2.33"];
      const d4 = commonStats[Math.floor(Math.random() * commonStats.length)];
      
      [d1, d2, d3, d4].forEach(item => {
        if (!pool.includes(item)) pool.push(item);
      });

      // Mezclar aleatoriamente el pool
      pool.sort(() => Math.random() - 0.5);

      pool.forEach(val => {
        const btn = document.createElement('button');
        btn.className = 'pool-item';
        btn.innerText = val;
        btn.onclick = () => selectFormulaValue(val);
        poolContainer.appendChild(btn);
      });
    }

    // Aproximación matemática para la CDF de la Normal Estándar N(0,1)
    function normalCDF(z) {
      const p = 0.3275911;
      const a1 = 0.254829592;
      const a2 = -0.284496736;
      const a3 = 1.421413741;
      const a4 = -1.453152027;
      const a5 = 1.061405429;
      
      const sign = (z < 0) ? -1 : 1;
      const x = Math.abs(z) / Math.sqrt(2.0);
      const t = 1.0 / (1.0 + p * x);
      const erf = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * Math.exp(-x * x);
      return 0.5 * (1.0 + sign * erf);
    }

    function setActivePValSlot(slotId) {
      activePValSlot = slotId;
      document.querySelectorAll('#phase-6 .math-slot').forEach(s => s.classList.remove('active-slot'));
      const activeSlotElement = document.getElementById(`slot-p-${slotId}`);
      if (activeSlotElement) {
        activeSlotElement.classList.add('active-slot');
      }

      const poolContainer = document.getElementById('pool-items-p-container');
      poolContainer.innerHTML = '';

      if (slotId.includes('op')) {
        const ops = ['≥', '<', '=', '>', '≤', '≠'];
        ops.forEach(op => {
          const btn = document.createElement('button');
          btn.className = 'pool-item';
          btn.innerText = op;
          btn.onclick = () => selectPValFormulaValue(op);
          poolContainer.appendChild(btn);
        });
      } else {
        const dist = activeExercise.dist;
        const calcValLabel = `${dist}<sub>calc</sub>`;
        const absCalcValLabel = `|${dist}<sub>calc</sub>|`;
        const critValLabel = `${dist}<sub>crit</sub>`;

        const distractors = [`μ₀`, `x̄`, `α` ];
        let pool = [calcValLabel, absCalcValLabel, critValLabel, ...distractors];

        pool = [...new Set(pool)];
        pool.sort(() => Math.random() - 0.5);

        pool.forEach(htmlLabel => {
          const btn = document.createElement('button');
          btn.className = 'pool-item';
          btn.innerHTML = htmlLabel;
          const cleanText = htmlLabel.replace(/<[^>]*>/g, '');
          btn.onclick = () => selectPValFormulaValue(cleanText);
          poolContainer.appendChild(btn);
        });
      }
    }

    function selectPValFormulaValue(value) {
      if (!activePValSlot) return;

      const slotElement = document.getElementById(`slot-p-${activePValSlot}`);
      slotElement.className = 'math-slot';

      const dist = activeExercise.dist;
      let expectedOp1 = '≥';
      let expectedOp2 = '<';
      let expectedVal1 = dist + 'calc';
      let expectedVal2 = dist + 'calc';

      if (activeExercise.direction === 'left') {
        expectedOp1 = '≤';
      } else if (activeExercise.direction === 'two-tailed') {
        expectedOp1 = '≥';
        expectedVal1 = '|' + dist + 'calc|';
        expectedVal2 = '|' + dist + 'calc|';
      }

      const expectedVal = activePValSlot === 'op1' ? expectedOp1
                        : activePValSlot === 'op2' ? expectedOp2
                        : activePValSlot === 'val1' ? expectedVal1
                        : expectedVal2;

      if (value === expectedVal) {
        let displayValue = value;
        if (value === dist + 'calc') {
          displayValue = `${dist}<sub>calc</sub>`;
        } else if (value === '|' + dist + 'calc|') {
          displayValue = `|${dist}<sub>calc</sub>|`;
        }
        slotElement.innerHTML = displayValue;
        slotElement.classList.add('correct-slot', 'filled');
        pSelections[activePValSlot] = value;

        let allFilled = Object.values(pSelections).every(v => v !== null);
        if (allFilled) {
          document.getElementById('btn-next-7').classList.remove('disabled');
          activePValSlot = null;
          document.querySelectorAll('#phase-6 .math-slot').forEach(s => s.classList.remove('active-slot'));
          document.getElementById('pool-items-p-container').innerHTML = '';
        } else {
          const order = activeExercise.direction === 'left' ? ['op1', 'val1'] : ['op1', 'val1', 'op2', 'val2'];
          const nextSlot = order.find(k => pSelections[k] === null);
          if (nextSlot) setActivePValSlot(nextSlot);
        }
      } else {
        slotElement.classList.add('wrong-slot');
        setTimeout(() => {
          slotElement.classList.remove('wrong-slot');
        }, 500);
      }
    }

    // GAUSSIAN CURVE
    function drawNormalCurve(phase, zCrit, zCalc) {
      const svg = document.getElementById(`normal-curve-svg-${phase}`);
      const mean = 200;
      const stdDev = 50;
      const scaleY = 220;

      function normalPdf(x) {
        return Math.exp(-0.5 * Math.pow((x - mean) / stdDev, 2));
      }

      let curvePoints = [];
      for (let x = 10; x <= 390; x++) {
        let y = 170 - normalPdf(x) * scaleY;
        curvePoints.push(`${x},${y}`);
      }
      const bgPath = document.getElementById(`curve-bg-path-${phase}`);
      bgPath.setAttribute('d', `M 10,170 L ${curvePoints.join(' L ')} L 390,170`);

      function zToX(z) { return mean + z * stdDev; }

      const rightPath = document.getElementById(`critical-zone-right-path-${phase}`);
      const rightLine = document.getElementById(`critical-threshold-right-line-${phase}`);
      if (rightPath) rightPath.style.display = 'none';
      if (rightLine) rightLine.style.display = 'none';

      const critPath = document.getElementById(`critical-zone-path-${phase}`);
      const critLine = document.getElementById(`critical-threshold-line-${phase}`);
      const infoText = document.getElementById(`plot-overlay-info-${phase}`);

      if (activeExercise.direction === 'left') {
        const xCrit = zToX(zCrit);
        const critPoints = [];
        critPoints.push(`10,170`);
        for (let x = 10; x <= xCrit; x++) {
          let y = 170 - normalPdf(x) * scaleY;
          critPoints.push(`${x},${y}`);
        }
        critPoints.push(`${xCrit},170`);
        critPath.setAttribute('d', `M ${critPoints.join(' L ')} Z`);
        if (critLine) {
          critLine.setAttribute('x1', xCrit);
          critLine.setAttribute('y1', 170 - normalPdf(xCrit) * scaleY);
          critLine.setAttribute('x2', xCrit);
          critLine.setAttribute('y2', 170);
        }

        if (infoText) {
          infoText.innerHTML = `<span style="color: var(--accent-red); font-weight: 800;">${activeExercise.dist}_crit = ${zCrit}</span><span>0</span>`;
        }
      } else if (activeExercise.direction === 'right') {
        const xCrit = zToX(zCrit);
        const critPoints = [];
        critPoints.push(`${xCrit},170`);
        for (let x = xCrit; x <= 390; x++) {
          let y = 170 - normalPdf(x) * scaleY;
          critPoints.push(`${x},${y}`);
        }
        critPoints.push('390,170');
        critPath.setAttribute('d', `M ${critPoints.join(' L ')} Z`);
        if (critLine) {
          critLine.setAttribute('x1', xCrit);
          critLine.setAttribute('y1', 170 - normalPdf(xCrit) * scaleY);
          critLine.setAttribute('x2', xCrit);
          critLine.setAttribute('y2', 170);
        }

        if (infoText) {
          infoText.innerHTML = `<span>0</span><span style="color: var(--accent-red); font-weight: 800;">${activeExercise.dist}_crit = ${zCrit}</span>`;
        }
      } else {
        const absCrit = Math.abs(zCrit);
        const xCritL = zToX(-absCrit);
        const xCritR = zToX(absCrit);

        const critPointsL = [];
        critPointsL.push(`10,170`);
        for (let x = 10; x <= xCritL; x++) {
          let y = 170 - normalPdf(x) * scaleY;
          critPointsL.push(`${x},${y}`);
        }
        critPointsL.push(`${xCritL},170`);
        critPath.setAttribute('d', `M ${critPointsL.join(' L ')} Z`);
        if (critLine) {
          critLine.setAttribute('x1', xCritL);
          critLine.setAttribute('y1', 170 - normalPdf(xCritL) * scaleY);
          critLine.setAttribute('x2', xCritL);
          critLine.setAttribute('y2', 170);
        }

        if (rightPath) rightPath.style.display = 'block';
        if (rightLine) rightLine.style.display = 'block';
        const critPointsR = [];
        critPointsR.push(`${xCritR},170`);
        for (let x = xCritR; x <= 390; x++) {
          let y = 170 - normalPdf(x) * scaleY;
          critPointsR.push(`${x},${y}`);
        }
        critPointsR.push(`390,170`);
        if (rightPath) rightPath.setAttribute('d', `M ${critPointsR.join(' L ')} Z`);
        if (rightLine) {
          rightLine.setAttribute('x1', xCritR);
          rightLine.setAttribute('y1', 170 - normalPdf(xCritR) * scaleY);
          rightLine.setAttribute('x2', xCritR);
          rightLine.setAttribute('y2', 170);
        }

        if (infoText) {
          infoText.innerHTML = `<span style="color: var(--accent-red); font-weight: 800;">-${absCrit}</span><span>0</span><span style="color: var(--accent-red); font-weight: 800;">+${absCrit}</span>`;
        }
      }

      if (phase === 6) {
        document.getElementById('region-subtitle-text').innerHTML = `Nivel de significación <span class="highlight-cyan">α = 0.05</span> en contraste <strong>${activeExercise.direction === 'two-tailed' ? 'bilateral' : 'unilateral'}</strong>.`;
        document.getElementById('region-crit-text').innerHTML = activeExercise.direction === 'two-tailed'
          ? `Rechazaremos $H_0$ si el valor observado es menor que $-${Math.abs(activeExercise.zCrit)}$ o mayor que $+${Math.abs(activeExercise.zCrit)}$.`
          : (activeExercise.direction === 'left' ? `Rechazaremos $H_0$ si el valor observado cae a la izquierda de $Z_{crit} = ${activeExercise.zCrit}$.` : `Rechazaremos $H_0$ si el valor observado cae a la derecha de $Z_{crit} = ${activeExercise.zCrit}$.`);
        triggerMathRender();
      }

      if (zCalc !== null && phase === 7) {
        const absCalc = Math.abs(zCalc);
        const xCalcL = zToX(-absCalc);
        const xCalcR = zToX(absCalc);

        const calcLine = document.getElementById(`z-calc-line-7`);
        const calcDot = document.getElementById(`z-calc-dot-7`);
        const calcLineRight = document.getElementById(`z-calc-right-line-7`);
        const calcDotRight = document.getElementById(`z-calc-right-dot-7`);

        if (calcLineRight) calcLineRight.style.display = 'none';
        if (calcDotRight) calcDotRight.style.display = 'none';

        const pPath = document.getElementById(`p-value-path-7`);
        const pPathRight = document.getElementById(`p-value-right-path-7`);
        if (pPathRight) pPathRight.style.display = 'none';

        if (activeExercise.direction === 'two-tailed') {
          // Bilateral: draw symmetric lines and dots
          calcLine.setAttribute('x1', xCalcL);
          calcLine.setAttribute('y1', 170 - normalPdf(xCalcL) * scaleY);
          calcLine.setAttribute('x2', xCalcL);
          calcLine.setAttribute('y2', 170);

          calcDot.setAttribute('cx', xCalcL);
          calcDot.setAttribute('cy', 170 - normalPdf(xCalcL) * scaleY);

          if (calcLineRight && calcDotRight) {
            calcLineRight.style.display = 'block';
            calcDotRight.style.display = 'block';
            calcLineRight.setAttribute('x1', xCalcR);
            calcLineRight.setAttribute('y1', 170 - normalPdf(xCalcR) * scaleY);
            calcLineRight.setAttribute('x2', xCalcR);
            calcLineRight.setAttribute('y2', 170);

            calcDotRight.setAttribute('cx', xCalcR);
            calcDotRight.setAttribute('cy', 170 - normalPdf(xCalcR) * scaleY);
          }

          // Shading both sides
          if (pPath) {
            const pPointsL = [];
            pPointsL.push(`10,170`);
            for (let x = 10; x <= xCalcL; x++) {
              let y = 170 - normalPdf(x) * scaleY;
              pPointsL.push(`${x},${y}`);
            }
            pPointsL.push(`${xCalcL},170`);
            pPath.setAttribute('d', `M ${pPointsL.join(' L ')} Z`);
          }

          if (pPathRight) {
            pPathRight.style.display = 'block';
            const pPointsR = [];
            pPointsR.push(`${xCalcR},170`);
            for (let x = xCalcR; x <= 390; x++) {
              let y = 170 - normalPdf(x) * scaleY;
              pPointsR.push(`${x},${y}`);
            }
            pPointsR.push(`390,170`);
            pPathRight.setAttribute('d', `M ${pPointsR.join(' L ')} Z`);
          }

        } else {
          // Unilateral
          const xCalc = zToX(zCalc);
          const yCalcVal = 170 - normalPdf(xCalc) * scaleY;

          calcLine.setAttribute('x1', xCalc);
          calcLine.setAttribute('y1', yCalcVal);
          calcLine.setAttribute('x2', xCalc);
          calcLine.setAttribute('y2', 170);

          calcDot.setAttribute('cx', xCalc);
          calcDot.setAttribute('cy', yCalcVal);

          const pPoints = [];
          if (activeExercise.direction === 'left') {
            pPoints.push(`10,170`);
            for (let x = 10; x <= xCalc; x++) {
              let y = 170 - normalPdf(x) * scaleY;
              pPoints.push(`${x},${y}`);
            }
            pPoints.push(`${xCalc},170`);
            pPath.setAttribute('d', `M ${pPoints.join(' L ')} Z`);
          } else {
            pPoints.push(`${xCalc},170`);
            for (let x = xCalc; x <= 390; x++) {
              let y = 170 - normalPdf(x) * scaleY;
              pPoints.push(`${x},${y}`);
            }
            pPoints.push(`390,170`);
            pPath.setAttribute('d', `M ${pPoints.join(' L ')} Z`);
          }
        }

        document.getElementById('calc-subtitle-text').innerHTML = `Estadístico observado: $${activeExercise.dist}_{calc} = ${activeExercise.zCalc}$ | p-valor = $${activeExercise.pValue}$.`;
        document.getElementById('calc-formula-math').innerHTML = `
          $$${activeExercise.dist}_{calc} = \\frac{${activeExercise.xbar} - ${activeExercise.mu0}}{${activeExercise.sigma} / \\sqrt{${activeExercise.n}}} = ${activeExercise.zCalc}$$
          Probabilidad acumulada bajo el supuesto nulo: $p\\text{-valor} = ${activeExercise.pValue}$.

      ];
      pool = [...new Set(pool)].sort(() => Math.random() - 0.5);

      pool.forEach(val => {
        const btn = document.createElement('button');
        btn.className = 'pool-item';
        btn.innerText = val;
        btn.onclick = () => selectDecValue(val);
        poolContainer.appendChild(btn);
      });
    }

    function selectDecValue(value) {
      if (!activeDecSlot) return;

      const direction = activeExercise.direction;
      const zCritVal = parseFloat(activeExercise.zCrit);
      
      let expectedVal = "";
      if (direction === 'left' || direction === 'right') {
        expectedVal = zCritVal.toString();
      } else {
        if (activeDecSlot === 'crit1') {
          expectedVal = (-Math.abs(zCritVal)).toString();
        } else {
          expectedVal = Math.abs(zCritVal).toString();
        }
      }

      const slotElement = document.getElementById(`slot-dec-${activeDecSlot}`);
      slotElement.className = 'math-slot';

      if (parseFloat(value) === parseFloat(expectedVal)) {
        slotElement.innerText = value;
        slotElement.classList.add('correct-slot', 'filled');
        decSelections[activeDecSlot] = value;

        if (direction === 'left') {
          document.getElementById('label-dec-crit2').innerText = value;
          document.getElementById('label-dec-crit2').className = 'highlight-red';
        } else if (direction === 'right') {
          document.getElementById('label-dec-crit2').innerText = value;
          document.getElementById('label-dec-crit2').className = 'highlight-red';
        } else {
          if (activeDecSlot === 'crit1') {
            document.getElementById('label-dec-crit3').innerText = value;
            document.getElementById('label-dec-crit3').className = 'highlight-red';
          } else {
            document.getElementById('label-dec-crit4').innerText = value;
            document.getElementById('label-dec-crit4').className = 'highlight-red';
          }
        }

        let allFilled = false;
        if (direction === 'two-tailed') {
          allFilled = decSelections['crit1'] !== null && decSelections['crit2'] !== null;
        } else {
          allFilled = decSelections['crit1'] !== null;
        }

        if (allFilled) {
          activeDecSlot = null;
          document.querySelectorAll('#phase-7 .math-slot').forEach(s => s.classList.remove('active-slot'));
          document.getElementById('pool-items-dec-container').innerHTML = '';
          document.getElementById('dec-pool-container-wrapper').style.display = 'none';
          
          document.getElementById('dec-membership-container').style.display = 'block';
        } else {
          if (activeDecSlot === 'crit1' && direction === 'two-tailed') {
            setActiveDecSlot('crit2');
          }
        }
      } else {
        slotElement.classList.add('wrong-slot');
        setTimeout(() => {
          slotElement.classList.remove('wrong-slot');
        }, 500);
      }
    }

    function selectMembership(regionType) {
      const isReject = activeExercise.direction === 'left' ? activeExercise.zCalc < activeExercise.zCrit
                     : activeExercise.direction === 'right' ? activeExercise.zCalc > activeExercise.zCrit
                     : Math.abs(activeExercise.zCalc) > Math.abs(activeExercise.zCrit);

      const expectedRegion = isReject ? 'critical' : 'acceptance';

      const choiceAccept = document.getElementById('choice-member-accept');
      const choiceReject = document.getElementById('choice-member-reject');

      choiceAccept.className = 'choice-card';
      choiceReject.className = 'choice-card';

      if (regionType === expectedRegion) {
        selectedMembership = regionType;
        if (regionType === 'acceptance') {
          choiceAccept.className = 'choice-card correct-selected';
        } else {
          choiceReject.className = 'choice-card correct-selected';
        }
        
        document.getElementById('dec-buttons-container').style.display = 'flex';
      } else {
        if (regionType === 'acceptance') {
          choiceAccept.className = 'choice-card wrong-selected';
        } else {
          choiceReject.className = 'choice-card wrong-selected';
        }
        setTimeout(() => {
          choiceAccept.className = 'choice-card';
          choiceReject.className = 'choice-card';
        }, 800);
      }
    }

    function makeFinalDecision(isReject) {
      const btnReject = document.getElementById('decision-reject');
      const btnAccept = document.getElementById('decision-accept');
      const feedback = document.getElementById('decision-feedback');
      const feedbackTitle = document.getElementById('decision-feedback-title');
      const feedbackText = document.getElementById('decision-feedback-text');
      const nextBtn = document.getElementById('btn-next-8');

      btnReject.className = 'btn-action';
      btnAccept.className = 'btn-action';

      let shouldReject = false;
      if (activeExercise.direction === 'left' && activeExercise.zCalc < activeExercise.zCrit) shouldReject = true;
      if (activeExercise.direction === 'right' && activeExercise.zCalc > activeExercise.zCrit) shouldReject = true;
      if (activeExercise.direction === 'two-tailed' && Math.abs(activeExercise.zCalc) > Math.abs(activeExercise.zCrit)) shouldReject = true;

      if (isReject === shouldReject) {
        if (isReject) {
          btnReject.className = 'btn-action correct-selected';
          btnReject.style.borderColor = 'var(--accent-green)';
          btnReject.style.boxShadow = 'var(--glow-green)';
        } else {
          btnAccept.className = 'btn-action correct-selected';
          btnAccept.style.borderColor = 'var(--accent-green)';
          btnAccept.style.boxShadow = 'var(--glow-green)';
        }

        feedback.className = "info-card green";
        feedbackTitle.innerText = "¡Decisión Correcta!";
        feedbackText.innerHTML = `Efectivamente debemos ${isReject ? '<strong>Rechazar H₀</strong>' : '<strong>No Rechazar H₀</strong>'}. El valor observado $${activeExercise.dist}_{calc} = ${activeExercise.zCalc}$ ${isReject ? 'pertenece a la región crítica (zona de rechazo)' : 'pertenece a la región de aceptación (no rechazo)'}.`;
        nextBtn.classList.remove('disabled');
      } else {
        if (isReject) btnReject.className = 'btn-action wrong-selected';
        else btnAccept.className = 'btn-action wrong-selected';

        feedback.className = "info-card red";
        feedbackTitle.innerText = "Revisa la decisión";
        feedbackText.innerHTML = `Dado que el estadístico pertenece a la región de ${selectedMembership === 'critical' ? 'rechazo' : 'aceptación'}, la decisión correspondiente debe alinearse con esto.`;
        nextBtn.classList.add('disabled');
      }
      feedback.style.display = 'block';
      triggerMathRender();

      document.getElementById('victory-hyp-text').innerHTML = `$\\mu_0 = ${activeExercise.mu0}$ frente a hipótesis alternante.`;
      document.getElementById('victory-verdict-text').innerHTML = shouldReject ? `Rechazar H₀. ${activeExercise.rejectMessage}` : `No Rechazar H₀. No hay suficiente evidencia estadística para respaldar la sospecha.`;
    }

    // Cajón de navegación
    function toggleDrawer(drawerId) {
      const drawer = document.getElementById(drawerId);
      const isCurrentlyOpen = drawer.classList.contains('open');
      document.querySelectorAll('.drawer').forEach(d => d.classList.remove('open'));
      if (!isCurrentlyOpen) drawer.classList.add('open');
      triggerMathRender();
    }

    // Auto-completar fase actual
    function autoCompleteCurrentPhase() {
      if (currentPhase === 0) {
        selectDistribution(activeExercise.dist);
      } else if (currentPhase === 1) {
        extSelections.mu0 = activeExercise.mu0;
        extSelections.sigma = activeExercise.sigma;
        extSelections.n = activeExercise.n;
        extSelections.xbar = activeExercise.xbar;
        
        document.getElementById('slot-ext-mu0').innerText = activeExercise.mu0;
        document.getElementById('slot-ext-mu0').className = 'extraction-slot correct-slot filled';
        document.getElementById('slot-ext-sigma').innerText = activeExercise.sigma;
        document.getElementById('slot-ext-sigma').className = 'extraction-slot correct-slot filled';
        document.getElementById('slot-ext-n').innerText = activeExercise.n;
        document.getElementById('slot-ext-n').className = 'extraction-slot correct-slot filled';
        document.getElementById('slot-ext-xbar').innerText = activeExercise.xbar;
        document.getElementById('slot-ext-xbar').className = 'extraction-slot correct-slot filled';

        const feedback = document.getElementById('ext-feedback');
        feedback.style.display = 'block';
        feedback.className = "info-card green";
        document.getElementById('ext-feedback-text').innerText = `¡Excelente! Has extraído todos los datos necesarios: media histórica (μ₀ = ${activeExercise.mu0}), desviación estándar (${activeExercise.dist === 'Z' ? 'σ' : 'S'} = ${activeExercise.sigma}), tamaño muestral (n = ${activeExercise.n}) y media observada (xbar = ${activeExercise.xbar}).`;
        document.getElementById('btn-next-1').classList.remove('disabled');
        activeExtSlot = null;
        document.querySelectorAll('.extraction-slot').forEach(s => s.classList.remove('active-slot'));
        document.getElementById('pool-items-ext-container').innerHTML = '';
      } else if (currentPhase === 2) {
        let h0OpExpected = '≥';
        let h1OpExpected = '<';
        if (activeExercise.direction === 'right') {
          h0OpExpected = '≤';
          h1OpExpected = '>';
        } else if (activeExercise.direction === 'two-tailed') {
          h0OpExpected = '=';
          h1OpExpected = '≠';
        }

        hypSelections['h0-op'] = h0OpExpected;
        hypSelections['h0-val'] = activeExercise.mu0;
        hypSelections['h1-op'] = h1OpExpected;
        hypSelections['h1-val'] = activeExercise.mu0;

        document.getElementById('slot-h0-op').innerText = h0OpExpected;
        document.getElementById('slot-h0-op').className = 'hyp-slot filled correct-slot';
        document.getElementById('slot-h0-val').innerText = activeExercise.mu0;
        document.getElementById('slot-h0-val').className = 'hyp-slot filled correct-slot';
        document.getElementById('slot-h1-op').innerText = h1OpExpected;
        document.getElementById('slot-h1-op').className = 'hyp-slot filled correct-slot';
        document.getElementById('slot-h1-val').innerText = activeExercise.mu0;
        document.getElementById('slot-h1-val').className = 'hyp-slot filled correct-slot';

        const feedback = document.getElementById('hyp-feedback');
        const feedbackText = document.getElementById('hyp-feedback-text');
        feedback.style.display = 'block';
        feedbackText.innerHTML = `Contraste <strong>${activeExercise.direction === 'left' ? 'unilateral izquierdo' : (activeExercise.direction === 'right' ? 'unilateral derecho' : 'bilateral')}</strong> para verificar ${activeExercise.claim}.`;
        document.getElementById('btn-next-2').classList.remove('disabled');
        activeHypSlot = null;
        document.querySelectorAll('.hyp-slot').forEach(s => s.classList.remove('selected-slot'));
        document.getElementById('pool-items-container').innerHTML = '';
        triggerMathRender();
      } else if (currentPhase === 3) {
        mathSelections.xbar = activeExercise.xbar;
        mathSelections.mu0 = activeExercise.mu0;
        mathSelections.n = activeExercise.n;

        document.getElementById('slot-f-xbar').innerText = activeExercise.xbar;
        document.getElementById('slot-f-xbar').className = 'math-slot correct-slot filled';
        document.getElementById('slot-f-mu0').innerText = activeExercise.mu0;
        document.getElementById('slot-f-mu0').className = 'math-slot correct-slot filled';
        document.getElementById('slot-f-n').innerText = activeExercise.n;
        document.getElementById('slot-f-n').className = 'math-slot correct-slot filled';

        const feedback = document.getElementById('param-feedback');
        const feedbackText = document.getElementById('param-feedback-text');
        feedback.style.display = 'block';
        feedbackText.innerHTML = `Sustitución exitosa en la fórmula del estadístico $${activeExercise.dist}$.`;
        document.getElementById('btn-next-3').classList.remove('disabled');
        activeMathSlot = null;
        document.querySelectorAll('.math-slot').forEach(s => s.classList.remove('active-slot'));
        document.getElementById('pool-items-formula-container').innerHTML = '';
        triggerMathRender();
      } else if (currentPhase === 4) {
        const dist = activeExercise.dist;
        const zCalcVal = activeExercise.zCalc;
        const absCalc = Math.abs(zCalcVal);
        
        let targetValue = "";
        const slot = document.getElementById('slot-p-lookup-val');
        if (slot) {
          if (dist === 'Z') {
            targetValue = normalCDF(absCalc).toFixed(4);
            const cells = document.querySelectorAll('#phase-4 .table-cell-val');
            cells.forEach(c => {
              const val = parseFloat(c.getAttribute('data-z'));
              if (Math.abs(val - absCalc) < 0.015) {
                c.classList.add('correct-cell');
                slot.innerHTML = c.innerHTML;
              }
            });
            if (slot.innerText === '?') {
              slot.innerText = targetValue;
            }
          } else {
            const df = parseInt(activeExercise.n) - 1;
            const cells = document.querySelectorAll('#phase-4 .table-cell-val');
            let matched = false;
            cells.forEach(c => {
              if (c.getAttribute('onclick') && c.getAttribute('onclick').includes(`'${df}'`)) {
                c.classList.add('correct-cell');
                slot.innerHTML = c.innerHTML;
                matched = true;
              }
            });
            if (!matched) {
              slot.innerText = absCalc.toString();
            }
          }
          slot.className = 'math-slot correct-slot filled';
        }
        selectedPValLookupValue = slot ? slot.innerText : absCalc.toString();
        document.getElementById('btn-next-4').classList.remove('disabled');
      } else if (currentPhase === 5) {
        selectContrastDir(activeExercise.direction);
        selectAlpha('0.05');
      } else if (currentPhase === 6) {
        selectedCritValue = activeExercise.zCrit.toString();
        const slot = document.getElementById('slot-crit-val');
        if (slot) {
          if (activeExercise.dist === 'Z') {
            const expectedAbs = Math.abs(parseFloat(activeExercise.zCrit));
            const cells = document.querySelectorAll('#phase-6 .table-cell-val');
            cells.forEach(c => {
              const val = parseFloat(c.getAttribute('data-z'));
              if (Math.abs(val - expectedAbs) < 0.015) {
                c.classList.add('correct-cell');
                slot.innerHTML = c.innerHTML;
              }
            });
          } else {
            const df = parseInt(activeExercise.n) - 1;
            const direction = activeExercise.direction;
            let expectedCol = 0;
            if (direction === 'two-tailed') expectedCol = 1;

            const cells = document.querySelectorAll('#phase-6 .table-cell-val');
            cells.forEach(c => {
              if (c.getAttribute('onclick') && c.getAttribute('onclick').includes(`'${df}'`) && c.getAttribute('onclick').includes(`, ${expectedCol}`)) {
                c.classList.add('correct-cell');
                slot.innerHTML = c.innerHTML;
              }
            });
          }
          if (slot.innerText === '?') {
            slot.innerText = activeExercise.zCrit;
          }
          slot.className = 'math-slot correct-slot filled';
        }
        drawNormalCurve(6, activeExercise.zCrit, null);
        document.getElementById('btn-next-6').classList.remove('disabled');
      } else if (currentPhase === 7) {
        const dist = activeExercise.dist;
        let expectedOp1 = '≥';
        let expectedOp2 = '<';
        let expectedVal1 = dist + 'calc';
        let expectedVal2 = dist + 'calc';

        if (activeExercise.direction === 'left') {
          expectedOp1 = '≤';
        } else if (activeExercise.direction === 'two-tailed') {
          expectedOp1 = '≥';
          expectedVal1 = '|' + dist + 'calc|';
          expectedVal2 = '|' + dist + 'calc|';
        }

        pSelections['op1'] = expectedOp1;
        pSelections['val1'] = expectedVal1;
        pSelections['op2'] = expectedOp2;
        pSelections['val2'] = expectedVal2;

        const slotOp1 = document.getElementById('slot-p-op1');
        const slotVal1 = document.getElementById('slot-p-val1');

        if (slotOp1) {
          slotOp1.innerText = expectedOp1;
          slotOp1.className = 'math-slot correct-slot filled';
        }
        if (slotVal1) {
          let displayVal1 = expectedVal1 === dist + 'calc' ? `${dist}<sub>calc</sub>` : `|${dist}<sub>calc</sub>|`;
          slotVal1.innerHTML = displayVal1;
          slotVal1.className = 'math-slot correct-slot filled';
        }
        
        if (activeExercise.direction !== 'left') {
          const slotOp2 = document.getElementById('slot-p-op2');
          const slotVal2 = document.getElementById('slot-p-val2');
          if (slotOp2) {
            slotOp2.innerText = expectedOp2;
            slotOp2.className = 'math-slot correct-slot filled';
          }
          if (slotVal2) {
            let displayVal2 = expectedVal2 === dist + 'calc' ? `${dist}<sub>calc</sub>` : `|${dist}<sub>calc</sub>|`;
            slotVal2.innerHTML = displayVal2;
            slotVal2.className = 'math-slot correct-slot filled';
          }
        }

        document.getElementById('btn-next-7').classList.remove('disabled');
        activePValSlot = null;
        document.querySelectorAll('#phase-7 .math-slot').forEach(s => s.classList.remove('active-slot'));
        document.getElementById('pool-items-p-container').innerHTML = '';
      } else if (currentPhase === 8) {
        const direction = activeExercise.direction;
        const zCritVal = parseFloat(activeExercise.zCrit);

        if (direction === 'left' || direction === 'right') {
          decSelections['crit1'] = zCritVal.toString();
          const slot = document.getElementById('slot-dec-crit1');
          if (slot) {
            slot.innerText = zCritVal.toString();
            slot.className = 'math-slot correct-slot filled';
          }
          const mirror = document.getElementById('label-dec-crit2');
          if (mirror) {
            mirror.innerText = zCritVal.toString();
            mirror.className = 'highlight-red';
          }
        } else {
          const c1Val = (-Math.abs(zCritVal)).toString();
          const c2Val = Math.abs(zCritVal).toString();
          decSelections['crit1'] = c1Val;
          decSelections['crit2'] = c2Val;
          const slot1 = document.getElementById('slot-dec-crit1');
          const slot2 = document.getElementById('slot-dec-crit2');
          if (slot1) {
            slot1.innerText = c1Val;
            slot1.className = 'math-slot correct-slot filled';
          }
          if (slot2) {
            slot2.innerText = c2Val;
            slot2.className = 'math-slot correct-slot filled';
          }
          const mirror1 = document.getElementById('label-dec-crit3');
          const mirror2 = document.getElementById('label-dec-crit4');
          if (mirror1) {
            mirror1.innerText = c1Val;
            mirror1.className = 'highlight-red';
          }
          if (mirror2) {
            mirror2.innerText = c2Val;
            mirror2.className = 'highlight-red';
          }
        }

        const wrapper = document.getElementById('dec-pool-container-wrapper');
        if (wrapper) wrapper.style.display = 'none';
        
        const memberContainer = document.getElementById('dec-membership-container');
        if (memberContainer) memberContainer.style.display = 'block';

        const isReject = direction === 'left' ? activeExercise.zCalc < activeExercise.zCrit
                       : direction === 'right' ? activeExercise.zCalc > activeExercise.zCrit
                       : Math.abs(activeExercise.zCalc) > Math.abs(activeExercise.zCrit);

        const expectedRegion = isReject ? 'critical' : 'acceptance';
        selectedMembership = expectedRegion;

        const choiceAccept = document.getElementById('choice-member-accept');
        const choiceReject = document.getElementById('choice-member-reject');
        if (choiceAccept && choiceReject) {
          choiceAccept.className = 'choice-card';
          choiceReject.className = 'choice-card';
          if (expectedRegion === 'acceptance') {
            choiceAccept.className = 'choice-card correct-selected';
          } else {
            choiceReject.className = 'choice-card correct-selected';
          }
        }

        const buttonsContainer = document.getElementById('dec-buttons-container');
        if (buttonsContainer) buttonsContainer.style.display = 'flex';

        makeFinalDecision(isReject);
      }
    }
